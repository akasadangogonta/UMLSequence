  /// <summary>
  /// ソート用の関数を取得.
  /// </summary>
  /// <returns>The present sort comparison.</returns>
  /// <param name="sortType">Sort type.</param>
  private System.Comparison<NwDataPresentList> GetPresentSortComparison( PresentSortType sortType )
  {
    switch ( sortType )
    {
    case PresentSortType.ExpiredNew:
      return (a,b)=>{ return ComparisonDate( a.expired_date, b.expired_date ); };
    case PresentSortType.ExpiredOld:
      return (a,b)=>{ return ComparisonDate( b.expired_date, a.expired_date ); };
	case PresentSortType.PresentNew:
		return (a,b)=>{ return ComparisonDate( a.present_date, b.present_date ); };
	case PresentSortType.PresentOld:
		return (a,b)=>{ return ComparisonDate( b.present_date, a.present_date ); };
    }
    return (a,b)=>{ return 0; };
  }
