  private void SavePushInfo()
  {
    cmdPushInfo = NetworkManager.Instance.GetCmd<NwCmdPushInfo>();
    cmdPushInfo.SetCallBack( this.SuccessPushInfo, this.FailedPushInfo );
    cmdPushInfo.ExecCmd( settingsBasic.isPush );
  }
  
重要、ここでUserDataManagerの変数にisPushを代入
  private void SuccessPushInfo()
  {
    base.FinishLoading();
    LoadingManager.Instance.Pause();
    UserDataManager.Instance.UpdatePushInfo( settingsBasic.isPush );
  }
  
  private void FailedPushInfo( NetworkError error )
  {
    LoadingManager.Instance.Pause();
    base.FinishLoading();
    MenuBase.ForceBreak();
    StartCoroutine( this.ReturnEditSettings() );
  }
