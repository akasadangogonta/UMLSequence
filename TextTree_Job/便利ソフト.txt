  public void SetTitleText( string format = "", bool isTextManager = true )
  {

	if( !isTextManager )
    {
	  this.titleName.Text = format;

	  float validWidth = Camera.main.pixelWidth * 0.600f;
	  if(backBtn.gameObject.activeSelf){
		validWidth = Camera.main.pixelWidth * 0.490f;		// ボタンがあるときはもう少し狭く
	  }
	  titleName.AdjustScaleX( validWidth );
      return;
    }
    if( !string.IsNullOrEmpty( format ))
    {
      string message = TextManager.GetText( format, TextType.TitleBoard );
      if( !string.IsNullOrEmpty( message ) )
      {
        this.titleName.Text = message;

		
		float validWidth = 0;
		if(backBtn.gameObject.activeSelf)
		{
			validWidth = Camera.main.pixelWidth * 0.400f;//0.490f;		// ボタンがあるときは少し狭く
		}else
		{
			validWidth = Camera.main.pixelWidth * 0.600f;//0.300f;
		}

		titleName.AdjustScaleX( validWidth );
      }
    }
    else
    {
      this.titleName.Text = "";
    }
  }
