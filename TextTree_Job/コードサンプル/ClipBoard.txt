  static public void SetClipBoard( string text )
  {

#if UNITY_EDITOR

    TextEditor te = new TextEditor();
    te.text = text;
    te.SelectAll();
    te.Copy();

#elif UNITY_ANDROID

    clipBoardManager = new AndroidJavaObject( "lh.plugin.clipboard.ClipBoardPlugin" );
    clipBoardManager.Call("copy", text);

#elif UNITY_IPHONE

    _PluginManager_Set_ClipBoard(text);

#endif
